65      /**
66       * Méthode permettant de completer un parking.
67       *
68       * @param parking parking à completer.
69       */
70      private void completeParking(Parking parking) {
71          Equipement equipement = getMapEquipements().get(parking.getIdObj());
72          if (equipement == null) {
73              LOGGER.warning("Pas d'equipements trouvés pour le parking " + parking.getIdentifiant());
74          } else {
75              parking.setLatitude(equipement.getLatitude());
76              parking.setLongitude(equipement.getLongitude());
77          }
78      }

85      /**
86       * @return {@link EquipementManager#mapEquipements}.

88      @SuppressWarnings("unchecked")
89      public Map<Integer, Equipement> getMapEquipements() {
90          if (mapEquipements == null) {
91              mapEquipements = new HashMap<Integer, Equipement>();
92              MoteurCsv moteurCsv = new MoteurCsv(Arrays.<Class<?>>asList(Equipement.class));
93              BufferedReader bufferedReader = null;
94              try {
95                  bufferedReader = new BufferedReader(new InputStreamReader(
96                          EquipementManager.class.getResourceAsStream(EQUIPEMENTS_PUBLICS_DEPLACEMENT_CSV), CHARSET_NAME));
97                  try {
98                      moteurCsv.parseFileAndInsert(bufferedReader, Equipement.class,
99                              new EquipementInsertObject(mapEquipements));
100                 } finally {
101                     try {
102                         bufferedReader.close();
103                     } catch (Exception exception) {
104                         LOGGER.warning(exception.getMessage());
105                     }
106                 }
107             } catch (UnsupportedEncodingException exception) {
108                 LOGGER.warning(exception.getMessage());
109             }
110 
111         }
112         return mapEquipements;
113     }
